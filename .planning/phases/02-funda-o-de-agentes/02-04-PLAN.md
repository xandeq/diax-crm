    ```csharp
    [HttpPost("{type}/chat")]
    public async Task<IActionResult> Chat(string type, [FromBody] AgentChatRequest body, CancellationToken ct)
    {
        if (!TryParseAgentType(type, out var agentType)) return BadRequest(new { message = "Tipo de agente inválido." });
        var userId = GetCurrentUserId(); if (!userId.HasValue) return Unauthorized();
        if (!await RbacGuardAsync(userId.Value, ct)) return StatusCode(403, new { message = "Acesso negado ao provider/modelo." });
        var sw = System.Diagnostics.Stopwatch.StartNew();
        var result = await _orchestrator.ChatAsync(agentType, userId.Value, body, ct);
        sw.Stop();
        if (result.IsSuccess) await LogAgentUsageAsync(userId.Value, agentType, result.Value, sw.Elapsed, success: true);
        return HandleResult(result);
    }

    [HttpPost("{type}/confirm")]
    public async Task<IActionResult> Confirm(string type, [FromBody] AgentConfirmRequest body, CancellationToken ct)
    {
        if (!TryParseAgentType(type, out var agentType)) return BadRequest(new { message = "Tipo de agente inválido." });
        var userId = GetCurrentUserId(); if (!userId.HasValue) return Unauthorized();
        if (!await RbacGuardAsync(userId.Value, ct)) return StatusCode(403, new { message = "Acesso negado ao provider/modelo." });
        var result = await _orchestrator.ConfirmAsync(agentType, userId.Value, body, ct);
        return HandleResult(result);
    }

    [HttpGet("{type}/conversations")]
    public async Task<IActionResult> ListConversations(string type, [FromQuery] int page = 1, [FromQuery] int pageSize = 20, CancellationToken ct = default)
    {
        if (!TryParseAgentType(type, out var agentType)) return BadRequest(new { message = "Tipo de agente inválido." });
        var userId = GetCurrentUserId(); if (!userId.HasValue) return Unauthorized();
        var result = await _orchestrator.ListConversationsAsync(agentType, userId.Value, page, pageSize, ct);
        return HandleResult(result);
    }

    [HttpGet("{type}/conversations/{id:guid}")]
    public async Task<IActionResult> GetConversation(string type, Guid id, CancellationToken ct)
    {
        if (!TryParseAgentType(type, out var agentType)) return BadRequest(new { message = "Tipo de agente inválido." });
        var userId = GetCurrentUserId(); if (!userId.HasValue) return Unauthorized();
        var result = await _orchestrator.GetConversationAsync(agentType, id, userId.Value, ct);
        return HandleResult(result);
    }
    ```
